class Palindrome:
    def check(self, num):
        temp = num
        reverse = 0
        while num > 0:
            rem = num % 10
            reverse = reverse * 10 + rem
            num = num // 10
        if temp == reverse:
            print("Palindrome Number")
        else:
            print("Not a Palindrome")

class Rectangle:

    def __init__(self, length, width):
        self.length = length
        self.width = width

    def area(self):
        return self.length * self.width

    def perimeter(self):
        return 2 * (self.length + self.width)

    def display(self):
        print("Area:", self.area())
        print("Perimeter:", self.perimeter())

=========10) if num even -> factorial and num odd -> find series=========
class NumberOperation:
 
    def __init__(self):
        self.num = int(input("Enter a number: "))
    def check(self):
        if self.num % 2 == 0:
            self.factorial()
        else:
            self.series()
    def factorial(self):
        fact = 1
        for i in range(1, self.num + 1):
            fact *= i
        print("Factorial is:", fact)
    def series(self):
        print("Series:", end=" ")
        for i in range(self.num, 0, -1):
            print(i, end=" ")
# Object creation
obj = NumberOperation()
obj.check()

class Account:
 
    def __init__(self):
        self.balance = 0
 
    def deposit(self, amount):
        self.balance += amount
        print("Amount Deposited:", amount)
        print("Current Balance:", self.balance)
 
    def withdraw(self, amount, min_balance):
        if self.balance - amount < min_balance:
            print("Cannot withdraw! Minimum balance must be maintained.")
        else:
            self.balance -= amount
            print("Amount Withdrawn:", amount)
            print("Current Balance:", self.balance)
 
    def interest(self, rate):
        interest_amount = self.balance * rate / 100
        print("Interest Amount:", interest_amount)
 
# Saving Account
class SavingAccount(Account):
 
    def __init__(self):
        super().__init__()
        self.min_balance = 1000
        self.rate = 10
 
 
# Current Account
